    var twoSum = function (nums, target) {
    for (let i = 0; i < nums.length; i++) {
        const num = nums[i];
        let l = i + 1;
        let r = nums.length - 1;
        while (l <= r) {
            const mid = Math.floor((l + r) / 2);
            if (num + nums[mid] > target) {
                r = mid - 1;
            } else if (num + nums[mid] < target) {
                l = mid + 1;
            } else {
                return [num, nums[mid]];
            }
        }
    }
};
